DROP DATABASE IF EXISTS smartcity;
CREATE DATABASE smartcity;
USE smartcity;

CREATE TABLE parking_lot(
    id INTEGER AUTO_INCREMENT PRIMARY KEY,
    name CHAR(50) UNIQUE NOT NULL,
    total_spots INT NOT NULL,
    latitude FLOAT(23, 20) NOT NULL,
    longitude FLOAT(23, 20) NOT NULL,
    description TEXT(1024) NOT NULL,
    hour_tax FLOAT(5, 2) NOT NULL
);

CREATE TABLE user(
    uuid UUID PRIMARY KEY,
    first_name CHAR(100) NOT NULL,
    last_name CHAR(100) NOT NULL,
    license_plate CHAR(10) NOT NULL,
    email CHAR(100) UNIQUE NOT NULL,
    password CHAR(100) NOT NULL,
    role enum('USER', 'ADMIN') NOT NULL
);

CREATE TABLE reservation(
                            uuid UUID PRIMARY KEY,
                            start_time DATETIME NOT NULL,
                            end_time DATETIME NOT NULL,
                            status ENUM('ACTIVE', 'CANCELLED', 'EXPIRED'),
                            id_parking_lot INTEGER NOT NULL,
                            id_user UUID NOT NULL,
                            CONSTRAINT fk_id_parking_lot
                                FOREIGN KEY (id_parking_lot)
                                    REFERENCES parking_lot(id),
                            CONSTRAINT fk_id_user
                                FOREIGN KEY (id_user)
                                    REFERENCES user(uuid)
);

CREATE INDEX indx_license_plate
    ON user(license_plate);
